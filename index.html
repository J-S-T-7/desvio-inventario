function exportarExcel() { // 1. Preparamos los datos de las filas const rows = data.map((p, i) => { const boxOdoo = valores[i].odoo / p.div; const dif = valores[i].real - boxOdoo; let estado = "CONFORME"; if (dif < -0.01) estado = "FALTANTE"; if (dif > 0.01) estado = "SOBRANTE"; return { "REFERENCIA": p.id, "PRODUCTO": p.nom, "FACTOR": p.div, "STOCK ODOO (UNID)": valores[i].odoo, "STOCK FÍSICO (CAJAS)": valores[i].real, "DIFERENCIA (CAJAS)": parseFloat(dif.toFixed(2)), "ESTADO": estado }; }); // 2. Creamos la hoja de trabajo (Worksheet) const ws = XLSX.utils.json_to_sheet(rows); // 3. Añadimos estilo básico a las columnas (Ancho) ws['!cols'] = [ { wch: 15 }, // Ref { wch: 40 }, // Producto { wch: 10 }, // Factor { wch: 15 }, // Odoo { wch: 15 }, // Físico { wch: 15 }, // Diferencia { wch: 15 } // Estado ]; // 4. Generamos el libro (Workbook) y descargamos const wb = XLSX.utils.book_new(); XLSX.utils.book_append_sheet(wb, ws, "Auditoría " + new Date().toLocaleDateString()); // Nombre del archivo con fecha actual const nombreArchivo = `Informe_Inventario_${new Date().toISOString().slice(0,10)}.xlsx`; XLSX.writeFile(wb, nombreArchivo); }
